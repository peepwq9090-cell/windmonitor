import requests import matplotlib.pyplot as plt import matplotlib.gridspec as gridspec from datetime import datetime import pytz # ============================== # ตั้งค่า ThingSpeak # ============================== CHANNEL_ID = "3279862" READ_API_KEY = "RW4WW5ICILK0P78L" FIELD = 1 # ============================== # ดึงข้อมูล 24 ชั่วโมง # ============================== url_24h = f"https://api.thingspeak.com/channels/{CHANNEL_ID}/fields/{FIELD}.json?api_key={READ_API_KEY}&results=200" response = requests.get(url_24h) data = response.json() feeds = data["feeds"] times_24h = [] values_24h = [] thai_tz = pytz.timezone("Asia/Bangkok") for f in feeds: if f[f"field{FIELD}"] is not None: utc_time = datetime.strptime(f["created_at"], "%Y-%m-%dT%H:%M:%SZ") thai_time = utc_time.replace(tzinfo=pytz.utc).astimezone(thai_tz) times_24h.append(thai_time) values_24h.append(float(f[f"field{FIELD}"])) # ============================== # ข้อมูล 1 ชั่วโมงล่าสุด # ============================== times_1h = times_24h[-60:] values_1h = values_24h[-60:] # ============================== # ค่าสูงสุดรายวัน (2 วันล่าสุด) # ============================== daily_max = {} for t, v in zip(times_24h, values_24h): day = t.strftime("%d/%m/%y") if day not in daily_max: daily_max[day] = [] daily_max[day].append(v) days = list(daily_max.keys())[-2:] max_values = [max(daily_max[d]) for d in days] # ============================== # ตรวจสถานะ (ถ้ามีข้อมูลล่าสุด) # ============================== if len(values_24h) > 0: system_status = "เปิด" status_color = "green" else: system_status = "ปิด" status_color = "red" # ============================== # สร้าง Layout ตามรูป # ============================== fig = plt.figure(figsize=(14,8)) fig.patch.set_facecolor('#d9d9d9') gs = gridspec.GridSpec(2, 2, width_ratios=[1,2], height_ratios=[1,1]) # -------- กราฟ 24 ชม -------- ax1 = fig.add_subplot(gs[0,0]) ax1.plot(times_24h, values_24h) ax1.set_title("กราฟแสดง 24 ชม.", fontsize=12, fontweight='bold') ax1.set_facecolor("white") ax1.grid(True) for spine in ax1.spines.values(): spine.set_linewidth(3) spine.set_color("black") # -------- กราฟ 1 ชม -------- ax2 = fig.add_subplot(gs[:,1]) ax2.plot(times_1h, values_1h) ax2.set_title("กราฟแสดงตลอด 1 ชม.", fontsize=12, fontweight='bold') ax2.set_facecolor("white") ax2.grid(True) for spine in ax2.spines.values(): spine.set_linewidth(3) spine.set_color("black") # -------- กราฟค่าสูงสุด -------- ax3 = fig.add_subplot(gs[1,0]) ax3.bar(days, max_values) ax3.set_title("กราฟต่อวันบันทึกค่าสูงสุด", fontsize=10, fontweight='bold') ax3.set_facecolor("white") ax3.grid(True) for spine in ax3.spines.values(): spine.set_linewidth(3) spine.set_color("black") # -------- หัวข้อใหญ่ -------- fig.suptitle("SARAPHI TECHNICAL WIND SPEED MONITOR TEST\n(กราฟแสดงผล ทดสอบ)", fontsize=18, fontweight="bold") # -------- สถานะ -------- fig.text(0.5, 0.02, f"สถานะการทำงาน : ( {system_status} )", ha='center', fontsize=14, color=status_color, fontweight="bold") plt.tight_layout(rect=[0,0.05,1,0.92]) plt.show()
